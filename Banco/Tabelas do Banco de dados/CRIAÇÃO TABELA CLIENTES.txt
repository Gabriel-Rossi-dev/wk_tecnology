CREATE TABLE CLIENTES (
    CODIGO INT NOT NULL UNIQUE,
    NOME VARCHAR(255),
    CIDADE VARCHAR(255),
    UF VARCHAR(255),
    ID_UF INT,
    PRIMARY KEY (CODIGO),
    INDEX idx_codigo (CODIGO),           
    INDEX idx_nome (NOME),               
    INDEX idx_cidade (CIDADE),           
    INDEX idx_uf (UF),                   
    INDEX idx_id_uf (ID_UF),             
    FOREIGN KEY (ID_UF) REFERENCES ESTADOS(ID)  
);
