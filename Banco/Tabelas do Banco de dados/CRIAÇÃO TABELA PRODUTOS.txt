    CODIGO INT NOT NULL UNIQUE,
    DESCRICAO VARCHAR(255),
    PRECO_VENDA FLOAT,
    PRIMARY KEY (CODIGO),
    INDEX idx_codigo (CODIGO),           
    INDEX idx_descricao (DESCRICAO),     
    INDEX idx_preco_venda (PRECO_VENDA)  
);
