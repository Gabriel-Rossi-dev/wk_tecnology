CREATE TABLE PEDIDOS_PRODUTOS (
    ID INT AUTO_INCREMENT PRIMARY KEY,
    NUMERO_PEDIDO INT,
    CODIGO_PRODUTO INT,
    QUANTIDADE INT,
    VALOR_UNITARIO FLOAT,
    VALOR_TOTAL FLOAT,
    INDEX idx_numero_pedido (NUMERO_PEDIDO),   
    INDEX idx_codigo_produto (CODIGO_PRODUTO), 
    INDEX idx_quantidade (QUANTIDADE),         
    INDEX idx_valor_unitario (VALOR_UNITARIO), 
    INDEX idx_valor_total (VALOR_TOTAL),       
    FOREIGN KEY (NUMERO_PEDIDO) REFERENCES PEDIDOS_DADOS_GERAIS(NUMERO_PEDIDO),
    FOREIGN KEY (CODIGO_PRODUTO) REFERENCES PRODUTOS(CODIGO)  
);
